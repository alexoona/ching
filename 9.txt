db.orderinfo.aggregate([
  { $match: { status: "A" } },
  {
    $group: {
      _id: "$status",
      avgPrice: { $avg: "$price" }
    }
  }
]);

db.orderinfo.find(
  { price: { $gte: 100, $lte: 1000 } },
  { status: 1, _id: 0 }
);

db.orderinfo.find({}, { _id: 0 });

db.orderinfo.createIndex({ price: 1 });
db.orderinfo.find({ price: { $gt: 200 } });
