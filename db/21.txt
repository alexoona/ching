SELECT *
FROM Duty_alloc
WHERE e_no = 123
AND shift = 'First'
AND month = 'April'
AND year = 2003;

SELECT *
FROM Empl
WHERE pay_rate >= (
    SELECT pay_rate
    FROM Empl
    WHERE e_name = 'Sachin'
);

CREATE VIEW salary_stats AS
SELECT post,
       MIN(pay_rate) AS min_salary,
       MAX(pay_rate) AS max_salary,
       AVG(pay_rate) AS avg_salary
FROM Empl
GROUP BY post;

SELECT d.shift, COUNT(DISTINCT d.e_no) AS total_employees
FROM Duty_alloc d
JOIN Empl e ON d.e_no = e.e_no
WHERE e.post = 'manager'
GROUP BY d.shift;
