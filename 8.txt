SELECT i.name, t.semester, t.year, t.course_id
FROM instructor i
JOIN teaches t ON i.T_ID = t.T_ID
ORDER BY t.year, t.semester;

CREATE VIEW student_view AS
SELECT S_ID, name, dept_name, tot_cred
FROM student;

ALTER TABLE student
CHANGE dept_name department_name VARCHAR(20);

DELETE FROM student
WHERE department_name IS NULL;
